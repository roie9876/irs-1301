```makefile
install:

run:
	npx -y concurrently -n "front,back" -c "cyan,green" \
		"cd frontend && npm run dev" \

```

**handleTest function:**
```typescript
async function handleTest() {
  setTestStatus("testing")
  setTestMessage("")
  try {
    const result = await api<{ success: boolean; message: string }>('/settings/test', {
      method: 'POST',
      body: JSON.stringify({
        provider: selectedProvider,
        model,
        api_key: apiKey,
        api_base: apiBase,
      }),
    })
    setTestStatus(result.success ? "success" : "error")
    setTestMessage(result.message)
  } catch (err) {
    setTestStatus("error")
    setTestMessage(err instanceof Error ? err.message : "שגיאה לא צפויה")
  }
}
```

**handleSave function:**
```typescript
async function handleSave() {
  setSaving(true)
  try {
    await api<SettingsResponse>('/settings', {
      method: 'POST',
      body: JSON.stringify({
        provider: selectedProvider,
        model,
        api_key: apiKey,
        api_base: apiBase,
      }),
    })
    setIsConfigured(true)
    setApiKey("") // Clear from memory after save
    setTestMessage("ההגדרות נשמרו בהצלחה ✓")
  } catch (err) {
    setTestStatus("error")
    setTestMessage(err instanceof ApiError ? err.message : "שגיאה בשמירה")
  } finally {
    setSaving(false)
  }
}
```

**Step 3: Update frontend/src/App.tsx** — add /settings route and first-launch redirect (per D-05):

```tsx
import { Routes, Route, Navigate } from 'react-router-dom'
import { useEffect, useState } from 'react'
import { AppLayout } from './components/layout/AppLayout'
import { SettingsPage } from './pages/SettingsPage'
import { api } from './lib/api'

interface SettingsStatus {
  has_api_key: boolean
  provider: string
}

function HomePage() {
  return (
    <div className="flex items-center justify-center min-h-[60vh]">
      <h1 className="text-2xl text-muted-foreground">ברוכים הבאים</h1>
    </div>
  )
}

export default function App() {
  const [settingsLoaded, setSettingsLoaded] = useState(false)
  const [isConfigured, setIsConfigured] = useState(true) // assume configured until checked

  useEffect(() => {
    api<SettingsStatus>('/settings')
      .then((s) => {
        setIsConfigured(s.has_api_key && !!s.provider)
        setSettingsLoaded(true)
      })
      .catch(() => {
        setIsConfigured(false)
        setSettingsLoaded(true)
      })
  }, [])

  if (!settingsLoaded) return null // brief loading state

  return (
    <AppLayout>
      <Routes>
        <Route
          path="/"
          element={isConfigured ? <HomePage /> : <Navigate to="/settings" replace />}
        />
        <Route path="/settings" element={<SettingsPage />} />
      </Routes>
    </AppLayout>
  )
}
```

The redirect logic: On first launch with no provider configured (per D-05), navigating to `/` redirects to `/settings`. Once settings are saved, `/` shows the home page.
